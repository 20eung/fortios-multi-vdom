config vdom
  edit VDOM-USsite

config vpn ipsec phase1-interface
    edit "ko-site-vpn"
        set interface "TEST21"
        set ike-version 2
        set keylife 28800
        set peertype any
        set net-device disable
        set proposal aes128-sha1
        set dpd on-idle
        set dhgrp 2
        set remote-gw 192.168.30.2
        set psksecret **netvpn
        set dpd-retryinterval 10
    next
end

config vpn ipsec phase2-interface
    edit "ko-site-vpn"
        set phase1name "ko-site-vpn"
        set proposal aes128-sha1
        set dhgrp 2
        set replay disable
        set auto-negotiate enable
        set keylifeseconds 3600
    next
end

config system interface
    edit "ko-site-vpn"
        set vdom "VDOM-USsite"
        set ip 169.254.34.1 255.255.255.255
        set allowaccess ping
        set type tunnel
        set remote-ip 169.254.34.2 255.255.255.255
        set tcp-mss 1240
        set interface "TEST21"
    next
end

config firewall policy
    edit 3
        set name "lo40-to-ko-site-vpn"
        set srcintf "loopback40"
        set dstintf "ko-site-vpn"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
    edit 4
        set name "ko-site-vpn-to-lo40"
        set srcintf "ko-site-vpn"
        set dstintf "loopback40"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "ALL"
        set logtraffic all
    next
end

config router static

    next
